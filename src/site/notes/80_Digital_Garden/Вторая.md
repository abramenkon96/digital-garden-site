# Для теста 2!
Это вторая заметка в моем гибридном хранилище.

<div id="three-container" style="width: 100%; height: 400px; border-radius: 12px; border: 1px solid var(--background-modifier-border); overflow: hidden; background: transparent;"></div>

<script type="module">
  // 1. Импортируем библиотеку Three.js напрямую через CDN
  import * as THREE from 'https://cdn.jsdelivr.net/npm/three@0.160.0/build/three.module.js';

  // 2. Находим наш контейнер в заметке
  const container = document.getElementById('three-container');

  // 3. Создаем базовые элементы сцены
  const scene = new THREE.Scene();
  
  // Создаем камеру (угол обзора, пропорции, ближняя/дальняя плоскость обрезки)
  const camera = new THREE.PerspectiveCamera(75, container.clientWidth / container.clientHeight, 0.1, 1000);
  camera.position.z = 2.5;

  // 4. Настраиваем рендерер
  // alpha: true делает фон прозрачным, чтобы он сливался с твоей темой сайта!
  const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
  renderer.setSize(container.clientWidth, container.clientHeight);
  container.appendChild(renderer.domElement);

  // 5. Создаем сам 3D-объект (Куб)
  const geometry = new THREE.BoxGeometry();
  // MeshNormalMaterial — крутой материал, который переливается цветами
  // в зависимости от угла обзора, ему даже не нужен источник света!
  const material = new THREE.MeshNormalMaterial();
  const cube = new THREE.Mesh(geometry, material);
  scene.add(cube);

  // 6. Функция анимации (вращение куба)
  function animate() {
    requestAnimationFrame(animate);
    
    // Скорость вращения
    cube.rotation.x += 0.01;
    cube.rotation.y += 0.01;
    
    renderer.render(scene, camera);
  }

  // Запускаем анимацию
  animate();

  // 7. Делаем так, чтобы сцена меняла размер, если посетитель сужает окно браузера или крутит телефон
  window.addEventListener('resize', () => {
    if(container) {
      camera.aspect = container.clientWidth / container.clientHeight;
      camera.updateProjectionMatrix();
      renderer.setSize(container.clientWidth, container.clientHeight);
    }
  });
</script>
