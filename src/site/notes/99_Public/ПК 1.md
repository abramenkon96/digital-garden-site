# Мой калькулятор



Здесь я решил сделать мини-приложение прямо внутри статьи.

<p id="result"></p>
<input type="text" id="calcInput" placeholder="Введите число">
<button onclick="calculate()">Умножить на 2</button>
<p id="result"></p>

<style>
  #calcInput { border: 2px solid var(--interactive-accent); padding: 5px; }
  button { background: var(--interactive-accent); color: white; border: none; padding: 5px 10px; cursor: pointer; }
</style>

<script>
  function calculate() {
    const val = document.getElementById('calcInput').value;
    document.getElementById('result').innerText = "Результат: " + (val * 2);
  }
</script>
<p id="result"></p>
